                 ¶               
               
Iterate over names and call the bake file method for each name. 
Apply callback when all files are baked.
             		                            
var bakeFiles = function(names, paths, callback) {
  var l, i, done;
  
  l = i = names.length;

  done = function(err) {
    if (--l === 0) {
      return callback(err); 
    } 
  };

  while (i--) {
    bakeFile(names[i], paths, done);
  } 
};
